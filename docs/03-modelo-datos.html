📊 Lectura IoT
🔒 Patrón Multi-Tenant — Row Level Security en todas las tablas
Todas las tablas del sistema tienen una columna tenant_id UUID NOT NULL con referencia a la tabla tenants. PostgreSQL RLS filtra automáticamente cada query por el tenant activo en la sesión, establecido al conectar vía SET app.current_tenant = '...'. Incluso con un bug de aplicación, es imposible que los datos de un tenant sean visibles para otro. 
-- Política RLS aplicada a todas las tablas (ejemplo: assets)
ALTER TABLE assets ENABLE ROW LEVEL SECURITY;

CREATE POLICY tenant_isolation ON assets
  USING (tenant_id = current_setting('app.current_tenant')::UUID);

-- El backend establece el tenant antes de cada query:
SET app.current_tenant = '{{tenant_uuid}}';

05 · Schema SQL — Tablas principales (Prisma-compatible)
tenants
Raíz multi-tenant
CREATE TABLE tenants (
  id           UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
  slug         TEXT        NOT NULL UNIQUE,
  name         TEXT        NOT NULL,
  plan         TEXT        NOT NULL DEFAULT 'starter',
  settings     JSONB       NOT NULL DEFAULT '{}',
  active       BOOLEAN     NOT NULL DEFAULT true,
  created_at   TIMESTAMPTZ NOT NULL DEFAULT now()
);
work_orders
Entidad central del sistema
CREATE TABLE work_orders (
  id              UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id       UUID        NOT NULL REFERENCES tenants(id),
  center_id       UUID        NOT NULL REFERENCES centers(id),
  code            TEXT        NOT NULL,        -- OT-2025-00341
  type            TEXT        NOT NULL,        -- corrective | preventive | soft_service…
  status          TEXT        NOT NULL DEFAULT 'draft',
  priority        TEXT        NOT NULL DEFAULT 'medium',
  title           TEXT        NOT NULL,
  description     TEXT,
  asset_id        UUID        REFERENCES assets(id),
  space_id        UUID        REFERENCES spaces(id),
  assigned_to     UUID        REFERENCES users(id),
  contract_id     UUID        REFERENCES contracts(id),
  pm_plan_id      UUID        REFERENCES pm_plans(id),
  sla_deadline    TIMESTAMPTZ,               -- calculado al crear
  started_at      TIMESTAMPTZ,
  closed_at       TIMESTAMPTZ,
  estimated_cost  NUMERIC(12,2),
  actual_cost     NUMERIC(12,2),
  metadata        JSONB        NOT NULL DEFAULT '{}',
  created_at      TIMESTAMPTZ  NOT NULL DEFAULT now(),
  updated_at      TIMESTAMPTZ  NOT NULL DEFAULT now(),
  UNIQUE(tenant_id, code)
);

-- RLS
ALTER TABLE work_orders ENABLE ROW LEVEL SECURITY;
CREATE POLICY tenant_isolation ON work_orders
  USING (tenant_id = current_setting('app.current_tenant')::UUID);

-- Índice principal dashboard
CREATE INDEX idx_wo_status_sla
  ON work_orders(tenant_id, status, sla_deadline)
  WHERE status NOT IN ('closed', 'cancelled');
sensor_readings
TimescaleDB Hypertable
CREATE TABLE sensor_readings (
  time        TIMESTAMPTZ NOT NULL,
  sensor_id   UUID        NOT NULL REFERENCES sensors(id),
  tenant_id   UUID        NOT NULL,
  value       NUMERIC     NOT NULL,
  unit        TEXT        NOT NULL,   -- "°C" | "kWh" | "pH" | "ppm"
  quality     TEXT        DEFAULT 'good'
);

-- Convertir a hypertable (TimescaleDB) particionada por tiempo
SELECT create_hypertable('sensor_readings', 'time',
  chunk_time_interval => INTERVAL '7 days');

-- Compresión automática de datos > 30 días
ALTER TABLE sensor_readings SET (
  timescaledb.compress,
  timescaledb.compress_segmentby = 'sensor_id'
);
SELECT add_compression_policy('sensor_readings', INTERVAL '30 days');
